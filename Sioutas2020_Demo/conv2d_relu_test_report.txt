-- Anderson2021 (CUDA) -------------------------------
auto pipeline = get_pipeline();
Func cr_relu_ac = pipeline.get_func(4);
Func cr_conv_ac = pipeline.get_func(3);
Var c(cr_relu_ac.get_schedule().dims()[0].var);
Var ci("ci");
Var n(cr_relu_ac.get_schedule().dims()[3].var);
Var x(cr_relu_ac.get_schedule().dims()[1].var);
Var xi("xi");
Var y(cr_relu_ac.get_schedule().dims()[2].var);
Var yi("yi");
Var yii("yii");
RVar r_x(cr_conv_ac.update(0).get_schedule().dims()[0].var);
RVar r_y(cr_conv_ac.update(0).get_schedule().dims()[1].var);
RVar r_z(cr_conv_ac.update(0).get_schedule().dims()[2].var);
Var ci_serial_outer("ci_serial_outer");
Var c_serial_outer("c_serial_outer");
Var xi_serial_outer("xi_serial_outer");
Var x_serial_outer("x_serial_outer");
Var yi_serial_outer("yi_serial_outer");
Var y_serial_outer("y_serial_outer");
cr_relu_ac
    .split(c, c, ci, 24, TailStrategy::ShiftInwards)
    .split(x, x, xi, 4, TailStrategy::ShiftInwards)
    .split(y, y, yi, 16, TailStrategy::ShiftInwards)
    .split(yi, yi, yii, 4, TailStrategy::ShiftInwards)
    .unroll(yii)
    .compute_root()
    .reorder(yii, ci, xi, yi, c, x, y, n)
    .gpu_blocks(c)
    .gpu_blocks(x)
    .fuse(y, n, y)
    .gpu_blocks(y)
    .split(ci, ci_serial_outer, ci, 24, TailStrategy::GuardWithIf)
    .gpu_threads(ci)
    .split(xi, xi_serial_outer, xi, 4, TailStrategy::GuardWithIf)
    .gpu_threads(xi)
    .split(yi, yi_serial_outer, yi, 4, TailStrategy::GuardWithIf)
    .gpu_threads(yi);
cr_conv_ac.update(0)
    .split(y, y, yi, 2, TailStrategy::GuardWithIf)
    .unroll(yi)
    .reorder(yi, r_x, r_y, r_z, c, x, y, n)
    .split(c, c_serial_outer, c, 24, TailStrategy::GuardWithIf)
    .gpu_threads(c)
    .split(x, x_serial_outer, x, 4, TailStrategy::GuardWithIf)
    .gpu_threads(x)
    .split(y, y_serial_outer, y, 8, TailStrategy::GuardWithIf)
    .gpu_threads(y);
cr_conv_ac
    .split(y, y, yi, 2, TailStrategy::ShiftInwards)
    .unroll(yi)
    .compute_at(cr_relu_ac, c)
    .reorder(yi, c, x, y, n)
    .split(c, c_serial_outer, c, 24, TailStrategy::GuardWithIf)
    .gpu_threads(c)
    .split(x, x_serial_outer, x, 4, TailStrategy::GuardWithIf)
    .gpu_threads(x)
    .split(y, y_serial_outer, y, 8, TailStrategy::GuardWithIf)
    .gpu_threads(y);
cr_conv_ac.in(cr_relu_ac).store_in(MemoryType::Register).compute_at(cr_relu_ac, ci)
    .bound_extent(c, 1)
    .unroll(c)
    .bound_extent(x, 1)
    .unroll(x)
    .bound_extent(y, 4)
    .unroll(y)
    .bound_extent(n, 1)
    .unroll(n);

-- Mullapudi2016 (CPU) -------------------------------
Var c_vi("c_vi");
Var c_vo("c_vo");

Func cr_conv_m = pipeline.get_func(3);
Func cr_relu_m = pipeline.get_func(4);

{
    Var c = cr_conv_m.args()[0];
    Var x = cr_conv_m.args()[1];
    RVar r$x(cr_conv_m.update(0).get_schedule().rvars()[0].var);
    RVar r$y(cr_conv_m.update(0).get_schedule().rvars()[1].var);
    RVar r$z(cr_conv_m.update(0).get_schedule().rvars()[2].var);
    cr_conv_m
        .compute_at(cr_relu_m, x)
        .split(c, c_vo, c_vi, 8)
        .vectorize(c_vi);
    cr_conv_m.update(0)
        .reorder(c, x, r$x, r$y, y, r$z, n)
        .split(c, c_vo, c_vi, 8, TailStrategy::GuardWithIf)
        .vectorize(c_vi);
}
{
    Var c = cr_relu_m.args()[0];
    Var y = cr_relu_m.args()[2];
    Var n = cr_relu_m.args()[3];
    cr_relu_m
        .compute_root()
        .split(c, c_vo, c_vi, 8)
        .vectorize(c_vi)
        .parallel(n)
        .parallel(y);
}
