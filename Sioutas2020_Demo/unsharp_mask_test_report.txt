-- Anderson2021 (CUDA) -------------------------------
auto pipeline = get_pipeline();
Func um_output_ac = pipeline.get_func(3);
Func um_sharpened_ac = pipeline.get_func(2);
Func um_blurred_ac = pipeline.get_func(1);
Var x(um_output_ac.get_schedule().dims()[0].var);
Var xi("xi");
Var y(um_output_ac.get_schedule().dims()[1].var);
Var yi("yi");
Var yii("yii");
RVar r_x(um_blurred_ac.update(0).get_schedule().dims()[0].var);
RVar r_y(um_blurred_ac.update(0).get_schedule().dims()[1].var);
Var xi_serial_outer("xi_serial_outer");
Var x_serial_outer("x_serial_outer");
Var yi_serial_outer("yi_serial_outer");
Var y_serial_outer("y_serial_outer");
um_output_ac
    .split(x, x, xi, 64, TailStrategy::ShiftInwards)
    .split(y, y, yi, 32, TailStrategy::ShiftInwards)
    .split(yi, yi, yii, 2, TailStrategy::ShiftInwards)
    .unroll(yii)
    .compute_root()
    .reorder(yii, xi, yi, x, y)
    .gpu_blocks(x)
    .gpu_blocks(y)
    .split(xi, xi_serial_outer, xi, 64, TailStrategy::GuardWithIf)
    .gpu_threads(xi)
    .split(yi, yi_serial_outer, yi, 16, TailStrategy::GuardWithIf)
    .gpu_threads(yi);
um_blurred_ac.update(0)
    .split(y, y, yi, 16, TailStrategy::GuardWithIf)
    .unroll(yi)
    .reorder(yi, r_x, r_y, x, y)
    .split(x, x_serial_outer, x, 64, TailStrategy::GuardWithIf)
    .gpu_threads(x)
    .split(y, y_serial_outer, y, 2, TailStrategy::GuardWithIf)
    .gpu_threads(y);
um_blurred_ac
    .split(y, y, yi, 16, TailStrategy::RoundUp)
    .unroll(yi)
    .compute_at(um_output_ac, x)
    .reorder(yi, x, y)
    .split(x, x_serial_outer, x, 64, TailStrategy::GuardWithIf)
    .gpu_threads(x)
    .split(y, y_serial_outer, y, 2, TailStrategy::GuardWithIf)
    .gpu_threads(y);

Var yi("yi");
Var yii("yii");

RVar r_y(um_blurred_a2.update(0).get_schedule().dims()[1].var);
um_output_a2
    .split(y, y, yi, 9, TailStrategy::ShiftInwards)
    .split(x, x, xi, 480, TailStrategy::ShiftInwards)
    .split(yi, yi, yii, 3, TailStrategy::ShiftInwards)
    .split(xi, xi, xii, 8, TailStrategy::ShiftInwards)
    .vectorize(xii)
    .compute_root()
    .reorder({xii, xi, yii, x, yi, y})
    .parallel(y);
um_sharpened_a2
    .store_in(MemoryType::Stack)
    .split(x, x, xi, 24, TailStrategy::ShiftInwards)
    .split(xi, xi, xii, 8, TailStrategy::ShiftInwards)
    .unroll(xi)
    .unroll(y)
    .vectorize(xii)
    .compute_at(um_output_a2, x)
    .reorder({xii, xi, y, x});
um_blurred_a2.update(0)
    .split(x, x, xi, 8, TailStrategy::GuardWithIf)
    .unroll(x)
    .unroll(y)
    .vectorize(xi)
    .reorder({xi, x, y, r_x, r_y});
um_blurred_a2
    .store_in(MemoryType::Stack)
    .split(x, x, xi, 8, TailStrategy::RoundUp)
    .unroll(x)
    .unroll(y)
    .vectorize(xi)
    .compute_at(um_sharpened_a2, x)
    .reorder({xi, x, y});


-- Mullapudi2016 (CPU) -------------------------------
Var x_vi("x_vi");
Var x_vo("x_vo");

Func um_blurred_m = pipeline.get_func(1);
Func um_output_m = pipeline.get_func(3);

{
    Var x = um_blurred_m.args()[0];
    Var y = um_blurred_m.args()[1];
    RVar r$x(um_blurred_m.update(0).get_schedule().rvars()[0].var);
    RVar r$y(um_blurred_m.update(0).get_schedule().rvars()[1].var);
    um_blurred_m
        .compute_at(um_output_m, y)
        .split(x, x_vo, x_vi, 8)
        .vectorize(x_vi);
    um_blurred_m.update(0)
        .reorder(r$x, x, r$y, y)
        .split(x, x_vo, x_vi, 8, TailStrategy::GuardWithIf)
        .vectorize(x_vi);
}
{
    Var x = um_output_m.args()[0];
    Var y = um_output_m.args()[1];
    um_output_m
        .compute_root()
        .split(x, x_vo, x_vi, 8)
        .vectorize(x_vi)
        .parallel(y);
}
